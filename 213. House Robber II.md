```python
class Solution:
    def rob(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        # similar to House Robber I, because we cannot have nms[0] and nums[-1] at the same time
        # it means, we have max(rob(nums[1:]), rob(nums[:-1]))
        
        # process rob[:-1] first
        prev_max = 0
        curr_max = 0
        for i in range(len(nums)-1):
            tmp = curr_max
            curr_max = max(curr_max, prev_max+nums[i])
            prev_max = tmp
        rec = curr_max
        
        # process rob[1ï¼š] 
        prev_max = 0
        curr_max = 0
        for i in range(1, len(nums)):
            tmp = curr_max
            curr_max = max(curr_max, prev_max+nums[i])
            prev_max = tmp
        return max(rec, curr_max)
```
Time complexity: O(N)  
Space complexity: O(1)
