```python
# DP with DP array
class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        # make sure dp[1] is valid
        if len(nums) < 3:
            return max(nums)
        dp = [0] * len(nums)
        dp[0] = nums[0]
        dp[1] = max(nums[0:2])
        for i in range(2, len(nums)):
            # transition function
            dp[i] = max(dp[i-2] + nums[i], dp[i-1])
        return dp[-1]
```
Time complexity:O(N)  
Space complexity: O(N)

```python
# DP without extra space
class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        # make sure two_ahead is valid
        if len(nums) < 3:
            return max(nums)
        two_ahead = nums[0]
        one_ahead =  max(nums[0], nums[1])
        current = 0 
        for i in range(2, len(nums)):
            # transition function
            current = max(two_ahead + nums[i], one_ahead)
            # update the maxsum of previous two variables
            two_ahead = one_ahead
            one_ahead = current
        return current
```
Time complexity:O(N)  
Space complexity: O(1)
